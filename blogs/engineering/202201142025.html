    # 命名

    • 普通变量：Camel，boolean类型的变量使用 is 或 has 开头；
    • 常量：全部大写字母，单词间以下划线_分割；
    • 私有属性、变量和方法：以下划线开头；
    • 函数：函数名和参数都用 Camel，函数名使用动宾短语；
    • 类：Pascal，使用名词；
    • 枚举变量：Pascal；
    • 枚举的属性：全部大写字母，单词间以下划线_分割；
    • 多个单词组成的缩写词，所有字母的大小写与首字母的大小写保持一致；
    • Promise 对象：使用动宾短语

    # 代码规范：

    1. 写代码之前一定要想好思路，尽量写出高质量的代码—少bug，安全性好。
      • 健壮性：基本的兼容性处理、边界处理、异常处理、用户输入校验，除非仅仅是内部函数
      • 可靠： 任何情况下，都返回一个可靠的结果
      • 宽容： 可维护性，输入可处理性等，考虑多种情况等
    2. 代码尽量简洁
    3. 对一些状态信息，用常量来定义，更易于代码阅读
    4. 写函数尽量加注释，按函数用途、参数解释的形式

    以随机数生成为例

    function isTypeOK(n) {
  return Number.isInteger(n) && n > 0;
}

function isRangeOK(n, min, max) {
  return n > 0 && n < max - min + 2;
}

function getRand(min, max) {
  return Math.floor(Math.random() * (max - min + 1)) + min;
}

function checkInArr(arr, num) {
  return arr.indexOf(num) !== -1;
}

function randomNum(n) {
  n = Math.floor(n);
  let min = 2;
  let max = 32;

  if (!isTypeOK(n) || !isRangeOK(n, min, max)) {
    console.error("请输入一个31以内的正整数！");
    return [];
  }
  let numArr = [];
  do {
    let randNum = getRand(min, max);
    if (!checkInArr(numArr, randNum)) {
      numArr.push(randNum);
    }
  } while (numArr.length < n);
  return numArr;
}

    # 解构

    • 使用对象解构的方式来访问多个属性值
    // bad

  const lastName = user.lastName;

  return `${firstName} ${lastName}`;
}

// good
function getFullName(user) {
  const { firstName, lastName } = user;

function getFullName({ firstName, lastName }) {
  return `${firstName} ${lastName}`;
}
