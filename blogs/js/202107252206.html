    # 随机数生成

    一般需要用到的工具函数有Math.random()，Math.floor()，Math.ceil()

    // 生成[0, 1)之间的随机数
function getRandom() {
  return Math.random()
}

// 生成两数之间的随机数
function getRandomArbitrary(min, max) {
  return Math.random() * (max - min) + min
}

// 获取两整数之间的随机整数
function getRandomInteger(min, max) {
  // return Math.floor(Math.random() * (max - min)) + min; //不含最大值，含最小值
  return Math.floor(Math.random() * (max - min + 1)) + min //含最大值，含最小值
}

    [1, 2, [3, [4, 5]], 6] => [1, 2, 3, 4, 5, 6]

    // 方法1 Array.prototype.flat()
const res1 = arr.flat(Infinity)

function isArray(arr) {
  return Array.isArray
    ? Array.isArray(arr)
    : Object.prototype.toString.call(val) === '[object Array]'
}

// 方法2 forEach + concat + isArray + recursivity
function deepFlat1(arr) {
  // 缓存递归结果
  let res = []
  arr.forEach((val) => {
    if (isArray(val)) res = res.concat(deepFlat(val))
    else res.push(val)
  })
  return res
}

// reduce + concat + isArray + recursivity
function deepFlat2(arr) {
  // 缓存递归结果
  return arr.reduce((acc, val) => {
    return acc.concat(isArray(val) ? deepFlat2(val) : val)
  }, [])
}

    # 查找兄弟单词

    兄弟单词：即只通过交换该单词的字母顺序，就能生成的单词，并且和原来的单词不同




     
     
     
     
     
     


    function isBrotherWord(w1, w2) {
  // 相同或长度不同，必定不是兄弟单词
  if (w1 === w2 || w1.length !== w2.length) return false
  const sortTemplate = (w) =>
    w
      .split('')
      .sort()
      .join()
  return sortTemplate(w1) === sortTemplate(w2)
}
